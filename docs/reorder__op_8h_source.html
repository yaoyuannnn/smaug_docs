48  } else if (output->getShape().ndims() == 2) {

50  }
51  } else if (input->getShape().ndims() == 3) {
52  // NTC->NCT or NCT->NTC.

54  targetLayout == DataLayout::NTC ||
55  srcLayout == DataLayout::NTC &&

97  TensorShape inferOutputShape() const {

99  if (inputShape.ndims() == 4 && (targetLayout == DataLayout::NC ||
100  targetLayout == DataLayout::CN)) {

102  std::vector<int> dims(2, 1);
103  dims[0] = inputShape[0];
104  for (int i = 1; i < inputShape.ndims(); ++i) {
105  dims[1] *= inputShape[i];

107  return TensorShape(dims, targetLayout, Backend::Alignment);
108  } else if (targetLayout == DataLayout::NC ||
109  targetLayout == DataLayout::CN) {
110  // Transpose a 2D tensor.
111  return TensorShape({ inputShape[1], inputShape[0] }, targetLayout,
112  Backend::Alignment);
113  } else if (targetLayout == DataLayout::NCT ||
114  targetLayout == DataLayout::NTC) {
115  // Transpose a 3D tensor.
116  return TensorShape({ inputShape[0], inputShape[2], inputShape[1] },
117  targetLayout, Backend::Alignment);
118  } else if (targetLayout == DataLayout::NCHW) {
119  return TensorShape({ inputShape[0], inputShape[3], inputShape[1],
120  inputShape[2] },
