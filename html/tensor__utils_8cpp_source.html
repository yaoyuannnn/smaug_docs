173 // Compute the tile size in this dimension with padding accounted for. The goal
174 // is to get the tile dimension size that doesn't have any elements unused,
175 // given the padding, weight and stride sizes.
176 //
177 // Args:
178 // maxTileDim: Maximum size of the tile size in this dimension.
179 // padding: Padding in the dimension.
180 // weightDim: Weight size in this dimension.
181 // stride: Stride size in this dimension.
182 // Returns:
183 // The tile size in this dimension.
184 int computePaddedTileDim(int maxTileDim,
185  int padding,
186  int weightDim,
187  int stride) {
188  // The number of strides we can take in this dimension.
189  int numStrides = (maxTileDim + padding - weightDim) / stride;
190  if (numStrides <= 0)
191  return maxTileDim;
192  int tileDim = weightDim + stride * numStrides;
193  return tileDim - padding;
194 }
195 } // namespace internal

197 TiledTensor generateTiledTensor(Tensor* tensor,
198  const TensorShape& tileShape,
199  Operator* op,
200  int fieldRows,
201  int fieldCols,
202  int rowStride,
203  int colStride,
204  PaddingType paddingType) {
205  const TensorShape& inputShape = tensor->getShape();
206  const int ndims = inputShape.ndims();
207  DataLayout layout = inputShape.getLayout();
208  // Compute the tiling halos. These are the rows/columns that the subsequent
209  // tile will overlap with the previous tile.
210  std::vector<int> tilingHalos(ndims, 0);
211  int hIdx = layout == NHWC ? 1 : NCHW ? 2 : -1;
212  int wIdx = layout == NHWC ? 2 : NCHW ? 3 : -1;
213  // The tilingHalos could be negative if fieldRows < rowStride, but we
214  // actually want that. For example, fieldRows = 1, rowStride = 2, then what
215  // the next tile wants is not to "borrow" any rows from the previous tile,
216  // but skip one row and start from there. So, -1 actually gives us the
217  // skipping effect.
218  if (hIdx != -1 && fieldRows != 0)
219  tilingHalos[hIdx] = fieldRows - rowStride;
220  if (wIdx != -1 && fieldCols != 0)
221  tilingHalos[wIdx] = fieldCols - colStride;
222  // Compute the input paddings.
223  int totalRowPad = (paddingType == SamePadding) ? fieldRows - 1 : 0;
224  int totalColPad = (paddingType == SamePadding) ? fieldCols - 1 : 0;
225  int topPad = FRAC_CEIL(totalRowPad, 2);
226  int leftPad = FRAC_CEIL(totalColPad, 2);
227  // This contains tile shapes in each dimension.
228  std::vector<std::vector<int>> tilesInDim(ndims);
229  // Compute the tile shapes in each dimension.
230  for (int i = 0; i < ndims; i++) {
231  int remaining = inputShape[i];
232  while (remaining > 0) {
233  int tileDim = std::min(tileShape[i], remaining);
234  bool firstTileInDim = tilesInDim[i].size() == 0;
235  bool lastTileInDim = remaining <= tileShape[i];
236  // Adjust the tile dimension size if we are at the first tile
237  // because of the top/left paddings.
238  if (i == hIdx && firstTileInDim && !lastTileInDim) {
239  tileDim = internal::computePaddedTileDim(
240  tileDim, topPad, fieldRows, rowStride);
241  } else if (i == wIdx && firstTileInDim && !lastTileInDim) {
242  tileDim = internal::computePaddedTileDim(
243  tileDim, leftPad, fieldCols, colStride);
244  }
245  tilesInDim[i].push_back(tileDim);
246  remaining -= tileDim;
247  if (remaining > 0)
248  remaining += tilingHalos[i];
249  }
250  }
251  std::vector<int> numBlocksInDim(ndims, 0);
252  for (int i = 0; i < ndims; i++)
253  numBlocksInDim[i] = tilesInDim[i].size();
254  TiledTensor tiledTensor(
255  TensorShape(numBlocksInDim, inputShape.getLayout()), tensor);
256  if (tiledTensor.size() == 1) {
257  // If there's only one tile, we don't need to tile the original tensor.
258  // So directly use it as the tile.
259  tiledTensor[0] = tensor;
260  } else {
261  std::vector<int> currentOrigin(ndims, 0);
262  for (auto tileIndex = tiledTensor.startIndex(); !tileIndex.end();
263  ++tileIndex) {
264  std::vector<int> currentTileShape(ndims);
265  for (int i = 0; i < ndims; i++)
266  currentTileShape[i] = tilesInDim[i][tileIndex.currentIndex(i)];
267  TensorShape currentShape(currentTileShape,
268  tileShape.getLayout(),
269  tileShape.getAlignment());
270  std::string tileName = op->getName() + ":" + tensor->getName() +
271  "/tile:" + std::to_string((int)tileIndex);
272  Tensor* tile = new Tensor(tileName, currentShape);
273  tile->allocateStorage(tensor->getDataType());
274  tiledTensor.setTile(tileIndex, currentOrigin, tile, false);
275  for (int i = ndims - 1; i >= 0; i--) {
276  currentOrigin[i] += currentShape[i];
277  if (currentOrigin[i] >= inputShape[i]) {
278  currentOrigin[i] = 0;
279  } else {
280  currentOrigin[i] -= tilingHalos[i];
281  break;
282  }
283  }
284  }
285  }
286  op->getWorkspace()->addTiledTensor(tiledTensor);
287  dout(1) << " Tiled Tensor " << tensor->getName() << ":\n"
288  << " original tensor shape: " << tensor->getShape() << "\n"
289  << " tile shape: " << tileShape
290  << ", number of tiles: " << tiledTensor.size() << "\n";
291  return tiledTensor;
292 }

324  // Copy data into the output tensor.
