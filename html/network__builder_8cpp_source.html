58 ActivationInfo getActivationInfo(const ActivationParams& params) {
59  ActivationInfo actInfo;
60  OpType opType = params.activation();
61  switch (opType) {
62  case OpType::ReLU:
63  actInfo.function = activation_type::RELU;
64  break;
65  case OpType::LReLU:
66  actInfo.function = activation_type::LRELU;
67  actInfo.params.slope = params.lrelu_params().slope();
68  break;
69  case OpType::ELU:
70  actInfo.function = activation_type::ELU;
71  actInfo.params.alpha = params.elu_params().alpha();
72  break;
73  case OpType::SELU:
74  actInfo.function = activation_type::SELU;
75  actInfo.params.alpha = params.elu_params().alpha();
76  actInfo.params.lambda = params.elu_params().lambda_param();
77  break;
78  case OpType::Tanh:
79  actInfo.function = activation_type::TANH;
80  break;
81  case OpType::HardTanh:
82  actInfo.function = activation_type::HARD_TANH;
83  actInfo.params.min = params.hard_tanh_params().min();
84  actInfo.params.max = params.hard_tanh_params().max();
85  break;
86  case OpType::Sigmoid:
87  actInfo.function = activation_type::SIGMOID;
88  break;
89  case OpType::Softmax:
90  actInfo.function = activation_type::SOFTMAX;
91  default:
92  actInfo.function = activation_type::NO_ACTIVATION;
93  }
94  return actInfo;
95 }

101  const TensorDataArray& tensorDataArray,
102  HostMemoryAccessPolicy memPolicy,
103  Network* network,
104  Workspace* workspace) {
105  const std::string& name = node.name();
106  OpType type = node.op();
107 
108  dout(0) << "Adding " << name << " (" << OpType_Name(type) << ").\n";
109 
110  if (type == OpType::Data) {
111  // Find the tensor data from the tensor data array.
112  TensorData tensorData;
113  for (int i = 0; i < tensorDataArray.data_array_size(); i++) {
114  if (tensorDataArray.data_array(i).name() ==
115  node.input_tensors(0).name()) {
116  tensorData = tensorDataArray.data_array(i);
117  break;
118  }
119  }
120  auto inputTensor = workspace->addTensor(
121  new Tensor(node.input_tensors(0), tensorData));
122  auto inputTensorOp = Backend::createDataOp(name, workspace);
123  inputTensorOp->setData(inputTensor);
124  network->addOperator(inputTensorOp);
125  } else if (type == OpType::Convolution3d ||
126  type == OpType::ConvolutionDepthwise) {
127  ConvolutionOp<Backend>* op;
128  if (type == OpType::Convolution3d)
129  op = Backend::createConvolutionOp(name, workspace);
130  else
131  op = Backend::createDepthwiseConvolutionOp(name, workspace);
132  assert(node.input_tensors_size() == 2);
133  const TensorProto& filterTensorProto = node.input_tensors(1);
134  const TensorShapeProto& shapeProto = filterTensorProto.shape();
135  assert(shapeProto.dims_size() == 4);
136  if (shapeProto.layout() == NCHW) {
137  op->setWeightDims(
138  shapeProto.dims(2), shapeProto.dims(3), shapeProto.dims(0));
139  } else {
140  op->setWeightDims(
141  shapeProto.dims(1), shapeProto.dims(2), shapeProto.dims(0));
142  }
143  const ConvParams& convParams = node.params().conv_params();
144  assert(convParams.stride_size() == 2);
145  op->setStride(convParams.stride(0), convParams.stride(1));
146  op->setPadding(convParams.padding());
147  op->setActivation(getActivationInfo(node.params().act_params()));
148  network->addOperator(op);
149  } else if (type == OpType::MaxPooling || type == OpType::AveragePooling) {
150  PoolingOp<Backend>* op;
151  if (type == MaxPooling)
152  op = Backend::createMaxPoolingOp(name, workspace);
153  else
154  op = Backend::createAvgPoolingOp(name, workspace);
155  const PoolParams& poolParams = node.params().pool_params();
156  assert(poolParams.stride_size() == 2);
157  assert(poolParams.pool_size_size() == 2);
158  op->setPoolingSize(poolParams.pool_size(0), poolParams.pool_size(1));
159  op->setPoolingStride(poolParams.stride(0), poolParams.stride(1));
160  network->addOperator(op);
161  } else if (type == OpType::InnerProduct) {
162  auto op = Backend::createInnerProductOp(name, workspace);
163  assert(node.input_tensors_size() == 2);
164  const TensorProto& weightTensorProto = node.input_tensors(1);
165  if (weightTensorProto.shape().layout() == NC)
166  op->setNumOutputs(weightTensorProto.shape().dims(0));
167  else
168  op->setNumOutputs(weightTensorProto.shape().dims(1));
169  op->setActivation(getActivationInfo(node.params().act_params()));
170  network->addOperator(op);
171  } else if (type == OpType::Reorder) {
172  DataLayout srcLayout = node.input_tensors(0).shape().layout();
173  DataLayout targetLayout = node.output_tensors(0).shape().layout();
174  ReorderOp<Backend>* op;
175  if (node.input_tensors(0).shape().dims_size() == 4 &&
176  (targetLayout == NC || targetLayout == CN)) {
177  op = Backend::createFlattenOp(name, workspace);
178  } else {
179  op = Backend::createReorderOp(name, workspace);
180  op->setTargetLayout(node.output_tensors(0).shape().layout());
181  }
182  network->addOperator(op);
183  } else if (type == OpType::Concat) {
184  auto op = Backend::createConcatOp(name, workspace);
185  op->setNumInputs(node.input_tensors_size());
186  op->setConcatAxis(node.params().concat_params().concat_axis());
187  network->addOperator(op);
188  } else if (type == OpType::Split) {
189  auto op = Backend::createSplitOp(name, workspace);
190  int axis = node.params().split_params().split_axis();
191  std::vector<int> splits;
192  for (const auto& tensor : node.output_tensors())
193  splits.push_back(tensor.shape().dims(axis));
194  op->setSplits(splits);
195  op->setSplitAxis(axis);
196  network->addOperator(op);
197  } else if (type == OpType::Reshape) {
198  auto op = Backend::createReshapeOp(name, workspace);
199  const TensorShapeProto& shapeProto = node.output_tensors(0).shape();
200  std::vector<int> shape(
201  shapeProto.dims().begin(), shapeProto.dims().end());
202  DataLayout layout = shapeProto.layout();
203  op->setShape(shape, layout);
204  network->addOperator(op);
205  } else if (type == OpType::Repeat) {
206  auto op = Backend::createRepeatOp(name, workspace);
207  const TensorShapeProto& inputShape = node.input_tensors(0).shape();
208  const TensorShapeProto& outputShape = node.output_tensors(0).shape();
209  std::vector<int> multiples;
210  for (int i = 0; i < inputShape.dims_size(); i++)
211  multiples.push_back(outputShape.dims(i) / inputShape.dims(i));
212  op->setMultiples(multiples);
213  network->addOperator(op);
214  } else if (type == OpType::BatchNorm) {
215  auto op = Backend::createBatchNormOp(name, workspace);
216  op->setActivation(getActivationInfo(node.params().act_params()));
217  network->addOperator(op);
218  } else if (type == OpType::EltwiseAdd) {
219  auto op = Backend::createEltwiseAddOp(name, workspace);
220  network->addOperator(op);
221  } else if (type == OpType::EltwiseMul) {
222  auto op = Backend::createEltwiseMulOp(name, workspace);
223  network->addOperator(op);
224  } else if (type == OpType::Less) {
225  auto op = Backend::createLessOp(name, workspace);
226  network->addOperator(op);
227  } else if (type == OpType::LessEqual) {
228  auto op = Backend::createLessEqualOp(name, workspace);
229  network->addOperator(op);
230  } else if (type == OpType::Greater) {
231  auto op = Backend::createGreaterOp(name, workspace);
232  network->addOperator(op);
233  } else if (type == OpType::GreaterEqual) {
234  auto op = Backend::createGreaterEqualOp(name, workspace);
235  network->addOperator(op);
236  } else if (type == OpType::Switch) {
237  auto op = Backend::createSwitchOp(name, workspace);
238  network->addOperator(op);
239  } else if (type == OpType::Merge) {
240  auto op = Backend::createMergeOp(name, workspace);
241  op->setNumInputs(node.input_tensors_size());
242  network->addOperator(op);
243  } else if (type == OpType::ReLU) {
244  auto op = Backend::createReluOp(name, workspace);
245  network->addOperator(op);
246  } else if (type == OpType::LReLU) {
247  // TODO: Add parameter to enable customization of this behavior.
248  auto op = Backend::createReluOp(name, workspace);
249  op->setSlope(0.1);
250  network->addOperator(op);
251  } else if (type == OpType::ELU) {
252  auto op = Backend::createEluOp(name, workspace);
253  network->addOperator(op);
254  } else if (type == OpType::SELU) {
255  auto op = Backend::createSeluOp(name, workspace);
256  network->addOperator(op);
257  } else if (type == OpType::Sigmoid) {
258  auto op = Backend::createSigmoidOp(name, workspace);
259  network->addOperator(op);
260  } else if (type == OpType::Softmax) {
261  auto op = Backend::createSoftmaxOp(name, workspace);
262  network->addOperator(op);
263  } else if (type == OpType::Tanh) {
264  auto op = Backend::createTanhOp(name, workspace);
265  network->addOperator(op);
266  } else if (type == OpType::HardTanh) {
267  auto op = Backend::createHardTanhOp(name, workspace);
268  network->addOperator(op);
269  } else if (type == OpType::UnknownOp) {
270  assert(false && "Invalid operator type!");
271  }
272 
273  Operator* op = network->getOperator(name);
274  // Set the sampling info for the operator if it supports sampling.
275  if (op->isSamplingSupported())
276  op->setSamplingInfo(network->getSamplingInfo());
277  // Set the memory access types for the operator's data.
278  if (memPolicy == HostMemoryAccessPolicy::AllDma) {
279  op->setInputsMemType(MemoryType::dma);
280  op->setWeightsMemType(MemoryType::dma);
281  op->setOutputsMemType(MemoryType::dma);
282  } else if (memPolicy == HostMemoryAccessPolicy::AllAcp) {
283  op->setInputsMemType(MemoryType::acp);
284  op->setWeightsMemType(MemoryType::acp);
285  op->setOutputsMemType(MemoryType::acp);
286  } else if (memPolicy == HostMemoryAccessPolicy::AllCache) {
287  op->setInputsMemType(MemoryType::cache);
288  op->setWeightsMemType(MemoryType::cache);
289  op->setOutputsMemType(MemoryType::cache);
290  } else if (memPolicy == HostMemoryAccessPolicy::AllAcpWithDmaForWeights) {
291  op->setInputsMemType(MemoryType::acp);
292  op->setWeightsMemType(MemoryType::dma);
293  op->setOutputsMemType(MemoryType::acp);
294  } else if (memPolicy == HostMemoryAccessPolicy::UnknownMemoryPolicy) {
295  assert(false && "Invalid host memory access policy!");
296  }

314 // Create the network by deserializing the graph stored in the
315 // protobuf model.
316 template <typename Backend>
317 static Network* createNetworkFromProto(const GraphProto& graphProto,
318  const TensorDataArray& tensorDataArray,
319  SamplingInfo& sampling,
320  Workspace* workspace) {
321  Network* network = new Network(graphProto.name());
322  network->setSamplingInfo(sampling);
323  for (int i = 0; i < graphProto.nodes_size(); i++) {
324  const NodeProto& node = graphProto.nodes(i);
325  createAndAddOperator<Backend>(node,
326  tensorDataArray,
327  graphProto.mem_policy(),
328  network,
329  workspace);
330  }
331 
332  // Now every operator has been added into the network, we can connect them
333  // together by adding edges in the graph view of the network.
334  for (int i = 0; i < graphProto.nodes_size(); i++) {
335  const NodeProto& node = graphProto.nodes(i);
336  Operator* op = network->getOperator(node.name());
337  for (int i = 0; i < node.parents_size(); i++) {
338  std::string inputOpName = node.parents(i);
339  int srcTensorIdx = node.src_tensors_indices(i);
340  Operator* inputOp = network->getOperator(inputOpName);
341  network->addEdge(inputOp, op, { srcTensorIdx, i });
342  }
343  }
344 
345  // Flowing through the graph edges (by doing a topological sort), we forward
346  // the output tensors of each operator (aka node) to its children.
347  const Graph& graph = network->getGraph();
348  EdgeNameMap edges = get(boost::edge_name, graph);
349  std::list<Vertex> vertices;
350  boost::topological_sort(graph, std::front_inserter(vertices));
351  for (auto v : vertices) {
352  Operator* op = get(boost::vertex_op, graph, v);
353  const std::vector<TensorBase*>& outputs = op->getOutputs();
354  out_edge_iter outEdgeIt, outEdgeEnd;
355  int srcIdx, destIdx;
356  for (boost::tie(outEdgeIt, outEdgeEnd) = out_edges(v, graph);
357  outEdgeIt != outEdgeEnd;
358  ++outEdgeIt) {
359  Vertex childVertex = target(*outEdgeIt, graph);
360  Operator* child = get(boost::vertex_op, graph, childVertex);
361  const TensorIndices& indices = edges[*outEdgeIt];
362  child->setInput(op->getOutput(indices.srcIdx), indices.destIdx);
363  }
364  }
365 
366  return network;
367 }

369 Network* smaug::buildNetwork(const std::string& modelTopo,
370  const std::string& modelParams,
371  SamplingInfo& sampling,
372  Workspace* workspace) {
373  // Parse the network topology from the protobuf text file.
374  GraphProto graph;
375  int modelTopoDescriptor = open(modelTopo.c_str(), O_RDONLY);
376  if (modelTopoDescriptor < 0) {
377  cout << modelTopo << ": network topology file not found." << endl;
378  exit(1);
379  }
380  google::protobuf::io::FileInputStream modelTopoInput(modelTopoDescriptor);
381  if (!google::protobuf::TextFormat::Parse(&modelTopoInput, &graph)) {
382  cout << "Failed to parse the network topology file!" << endl;
383  exit(1);
384  }
385  // Parse the network parameters from the protobuf binary file.
386  TensorDataArray tensorDataArray;
387  fstream modelParamsFile(modelParams, ios::in | ios::binary);
388  if (!modelParamsFile) {
389  cout << modelParams << ": network parameters file not found." << endl;
390  exit(1);
391  } else if (!tensorDataArray.ParseFromIstream(&modelParamsFile)) {
392  cout << "Failed to parse the network parameters file.\n";
393  exit(1);
394  }
395 
396  cout << "======================================================\n";
397  cout << " Loading the network model...\n";
398  cout << "======================================================\n";
399  Network* network = nullptr;
400  if (graph.backend() == ReferenceBackend::Name) {
401  network = createNetworkFromProto<ReferenceBackend>(
402  graph, tensorDataArray, sampling, workspace);
403  } else if (graph.backend() == SmvBackend::Name) {
404  network = createNetworkFromProto<SmvBackend>(
405  graph, tensorDataArray, sampling, workspace);
406  } else {
407  assert(false && "Unknown backend!");
408  }
409 
410  cout << "======================================================\n";
411  cout << " Summary of the network.\n";
412  cout << "======================================================\n";
413  network->printSummary();
414  return network;
415 }
